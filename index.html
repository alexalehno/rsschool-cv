      function getMatrixElementsSum(matrix) {
        let sum = 0;
        for (let i = 0; i < matrix.length; i++) {
          next: for (let j = 0; j < matrix[i].length; j++) {
            if (matrix[i - 1]) {
              if (matrix[i - 1][j] === 0) {
                continue next;
              }
            }
            sum += matrix[i][j];
          }
        }
        return sum;
      }
